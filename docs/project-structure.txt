api/
│── Controllers/                      // Chứa các controller xử lý request từ client
│   ├── AccountController.cs          // Xử lý đăng ký, đăng nhập, quản lý tài khoản
│   ├── RoleController.cs             // Xử lý quản lý roles (Admin, User, Owner)
│
├── Data/                             // Chứa lớp làm việc với cơ sở dữ liệu
│   ├── ApplicationDbContext.cs       // DbContext của Entity Framework để quản lý database
│
├── Dtos/                             // Chứa các lớp DTO để truyền dữ liệu giữa client và server
│   ├── Account/
│   │   ├── LoginDto.cs               // DTO cho yêu cầu đăng nhập
│   │   ├── RegisterDto.cs            // DTO cho yêu cầu đăng ký
│   │   ├── UserDto.cs                // DTO trả về thông tin user
│
├── Extensions/                       // Chứa các phương thức mở rộng (extension methods)
│   ├── ClaimsExtensions.cs           // Mở rộng Claims để lấy thông tin user từ token
│
├── Helpers/                          // Chứa các class hỗ trợ xử lý logic
│   ├── JwtTokenHelper.cs             // Hỗ trợ tạo và xác minh JWT token
│
├── Interfaces/                       // Chứa các interface định nghĩa hợp đồng cho service/repository
│   ├── IAuthService.cs               // Interface cho service xác thực
│   ├── ITokenService.cs              // Interface cho service xử lý JWT token
│
├── Mappers/                          // Chứa các lớp ánh xạ dữ liệu giữa Model và DTO
│   ├── UserMapper.cs                 // Chuyển đổi giữa User model và UserDto
│
├── Migrations/                       // Chứa các file migration của Entity Framework
│   ├── ...                           // Các file migration được tự động tạo ra khi cập nhật database
│
├── Models/                           // Chứa các entity chính của ứng dụng
│   ├── AppUser.cs                    // Mô tả người dùng (User) trong hệ thống
│
├── Properties/                       // Chứa file cấu hình môi trường chạy ứng dụng
│   ├── launchSettings.json           // Cấu hình môi trường chạy ứng dụng khi debug
│
├── Repository/                       // Chứa các lớp repository thực thi truy vấn dữ liệu
│   ├── AuthRepository.cs             // Repository xử lý đăng nhập, đăng ký
│
├── Service/                          // Chứa các service xử lý logic nghiệp vụ
│   ├── AuthService.cs                // Service xử lý xác thực
│   ├── TokenService.cs               // Service tạo và xác minh JWT token
│
├── .gitignore                        // Loại trừ các file không cần thiết khỏi git
├── Program.cs                        // Điểm khởi động của ứng dụng
├── api.csproj                        // File cấu hình dự án ASP.NET Core
├── appsettings.Development.json      // Cấu hình ứng dụng trong môi trường phát triển
└── appsettings.json                  // Cấu hình chung của ứng dụng (database, JWT, v.v.)
